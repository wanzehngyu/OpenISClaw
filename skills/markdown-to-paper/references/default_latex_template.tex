% ── Section heading styles ────────────────────────────────────────────────
\makeatletter
\def\@sect#1#2#3#4#5#6[#7]{%
  \ifnum #2>\@ne
    \refstepcounter{#1}%
    \ifnum #2<4
      \typeout{#1\space\arabic{#1}}%
    \fi
  \fi
  \par\nobreak
  \ifnum #2>\@ne \newpage\fi
  \@tempskipa #3\relax
  \ifdim \@tempskipa>\z@
    \begingroup
      #6{%
        \@hangfrom{\hskip #4\relax}%
          \interlinepenalty \@M #7\@@par}%
    \endgroup
    \csname #1mark\endcsname{#7}%
    \addcontentsline{toc}{#1}{\ifnum #2>\@ne\protect\numberline{\csname the#1\endcsname}\fi#7}%
  \else
    \def\@svsecdd{%
      \addpenalty{\@secpenalty}%
      \addvspace{#3}%
      \@ifstar{%
        \edef\@svsec{\toclevel@#1{\@secnumfont{\csname the#1\endcsname}}}%
      }{%
        \edef\@svsec{\toclevel@#1{\@secnumfont{\csname the#1\endcsname}}}%
      }%
      \protected@edef\@svsec{\@svsec\enspace #7}%
      \unskip\@svsec\@@par
    }%
  \fi
  \ifnum #2>\@ne
    \@xsect{#3}%
  \fi}
\makeatother
